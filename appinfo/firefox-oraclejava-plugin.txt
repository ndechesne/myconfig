1. download Oracle java (latest jdk)

2. unpack into /opt
	# cd /opt
	# gzip -d < /home/trevor/Downloads/jdk...tar.gz | tar xfv -

3. update alternatives
	# update-alternatives /usr/bin/java java /opt/jdk.../bin/java 1
	# update-alternatives /usr/bin/javac javac /opt/jdk.../bin/javac 1
	# update-alternatives /usr/bin/javadoc javadoc /opt/jdk.../bin/javadoc 1
	.
	.
	.
	jdb, jar, javaws

4. update firefox browser plugin
	# cd /usr/lib64/browser-plugins
	# rm libnpjp2.so
	# ln -s /opt/jdk.../jre/lib/libnpjp2.so

5. restart firefox

6. check
	url -> about:plugins
	http://java/com/en/download/installed.jsp
